def plot_hist_pdf_cdf(title, tuile_dpe, ardoise_dpe):
    """ Create figure with histogram, PDFs and CDFs """  
    fig, ((ax1, ax2, ax3), 
          (ax4, ax5, ax6), 
          (ax7, ax8, ax9)) = plt.subplots(nrows=3, ncols=3, sharex=True, figsize=(10,5))
    
    ## Collecting data to plot
    # number of bins in histogram
    data_tuile = tuile_dpe.adedpe202006_logtype_ph_u
    data_ardoise = ardoise_dpe.adedpe202006_logtype_ph_u
        #     k_tuile = sqrt(len(data_tuile))       # sqrt(n) rule
        #     k_ardoise = sqrt(len(data_ardoise))   # sqrt(n) rule
    k_tuile = (data_tuile.max() - data_tuile.min()) / freedman_bin_width(data_tuile)         # Freedman-Diaconis rule
    k_ardoise = (data_ardoise.max() - data_ardoise.min()) / freedman_bin_width(data_ardoise) # Freedman-Diaconis rule
    v_bins = ceil(mean([k_tuile,k_ardoise]))
    print("Nb of bins:", v_bins)
        #     v_bins = "auto"  # auto define bins for each dataset, different for each
    # histogram data
    # - tuile
    density_tuile, bins_tuile = np.histogram(data_tuile, bins=v_bins)
    unity_density_tuile = density_tuile / density_tuile.sum()    
    # - ardoise
    density_ardoise, bins_ardoise = np.histogram(data_ardoise, bins=v_bins)
    unity_density_ardoise = density_ardoise / density_ardoise.sum()
    
    ## Plotting
    # plot tuile
    widths_tuile = bins_tuile[:-1] - bins_tuile[1:]
    format_tui = {"width":widths_tuile, "color":"firebrick", "alpha":0.7, "label":'Tuile'}
    ax1.bar(bins_tuile[1:], density_tuile, **format_tui)
    ax2.bar(bins_tuile[1:], unity_density_tuile,**format_tui)
    ax3.bar(bins_tuile[1:], unity_density_tuile.cumsum(), **format_tui)
    # plot ardoise
    widths_ardoise = bins_ardoise[:-1] - bins_ardoise[1:]
    format_ard = {"width":widths_ardoise, "color":"cadetblue", "alpha":0.6, "label":'Ardoise'}
    ax4.bar(bins_ardoise[1:], density_ardoise, **format_ard)
    ax5.bar(bins_ardoise[1:], unity_density_ardoise, **format_ard)
    ax6.bar(bins_ardoise[1:], unity_density_ardoise.cumsum(), **format_ard)
    # plot both
    #  tuile
    ax7.bar(bins_tuile[1:], density_tuile, **format_tui)
    ax8.bar(bins_tuile[1:], unity_density_tuile, **format_tui)
    ax9.bar(bins_tuile[1:], unity_density_tuile.cumsum(), **format_tui)
    # - ardoise
    ax7.bar(bins_ardoise[1:], density_ardoise, **format_ard)
    ax8.bar(bins_ardoise[1:], unity_density_ardoise, **format_ard)
    ax9.bar(bins_ardoise[1:], unity_density_ardoise.cumsum(), **format_ard)
    
    ## Formatting
    # Y axis labels
    ax1.set_ylabel(f'Tuile \n total : {len(data_tuile)}')
    ax4.set_ylabel(f'Ardoise \n total : {len(data_ardoise)}')
    ax7.set_ylabel('Combine')
    # X axis labels
    ax7.set_xlabel('Histogrammes')
    ax8.set_xlabel('PDFs')
    ax9.set_xlabel('CDFs')
    # title
    ax2.set_title(f'U-valeur plancher haut [W/m/K]\n{title} ')
    
    fig.tight_layout()
    plt.legend(loc=4)
    fig.savefig(fr"plot_hist_pdf_cdf\{title}_hist_pdf_cdf.png")
